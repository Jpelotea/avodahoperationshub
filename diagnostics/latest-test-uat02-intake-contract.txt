225:function publicConstantTimeEquals_(left, right) {
226:  var a = String(left || '');
227:  var b = String(right || '');
228:  var mismatch = a.length ^ b.length;
229:  var length = Math.max(a.length, b.length);
230:  for (var index = 0; index < length; index += 1) {
231:    mismatch |= (a.charCodeAt(index % Math.max(a.length, 1)) || 0) ^
232:      (b.charCodeAt(index % Math.max(b.length, 1)) || 0);
233:  }
234:  return mismatch === 0;
235:}
